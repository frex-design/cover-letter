-- 送り先マスタ CREATE TABLE recipients ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, company_name text NOT NULL, department text, contact_name text, created_at timestamptz DEFAULT now() ); -- 差出人情報 CREATE TABLE sender_info ( id int PRIMARY KEY DEFAULT 1, company_name text, postal_code text, address text, building text, phone text, email text, department text, contact_name text ); -- 送付書 CREATE TABLE cover_letters ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, recipient_id uuid REFERENCES recipients(id), subject text, body_text text, created_at timestamptz DEFAULT now(), updated_at timestamptz DEFAULT now() ); -- 送付内容 CREATE TABLE cover_letter_items ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, cover_letter_id uuid REFERENCES cover_letters(id) ON DELETE CASCADE, description text, document_type text, quantity int DEFAULT 1, sort_order int DEFAULT 0 );
